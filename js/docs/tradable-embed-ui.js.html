    Welcome to the tradable-embed documentation. Tradable Embed is an open API to easily build financial trading features into any app, through any brokerage. This is the documentation for the tradable-embed javascript SDK, which is split into 2 components:

    • tradable-embed: See here
    • tradable-embed-ui: Provides Widgets and the website scraper - plug and play! (requires 'tradable-embed')

    In order to use tradable-embed-ui include the following scripts in your site (we use jQuery in no coflict mode and we assign it to the variable 'trEmbJQ'):

    <script type="text/javascript" src="//ajax.googleapis.com/ajax/libs/jquery/2.1.4/jquery.min.js"></script>
    <script type="text/javascript" id="tradable-embed" src="//js-api.tradable.com/1.2/tradable-embed.min.js" data-app-id="{your_app_id}"></script>
    <script type="text/javascript" id="tradable-embed-ui" src="//js-api.tradable.com/1.2/tradable-embed-ui.min.js"></script>

    NOTE! The oauth flow doesn't work properly in this documentation site for the codepen examples, for the best experience please see the examples on Codepen.io, you can do that clicking on the "Edit on Codepen" link

    Bottom Banner

    Adding the Tradable Embed UI script to your page will automatically add our bottom banner with the trader's dashboard & portfolio. If you want to turn it off, just add the class 'tr-embed-bottom-banner-off' to the Embed UI script

    <script type="text/javascript" id="tradable-embed-ui" src="//js-api.tradable.com/1.2/tradable-embed-ui.min.js" class="tr-embed-bottom-banner-off"></script>

    promptOrderEntry

    method
    tradableEmbed.promptOrderEntry()

    Option name Type Description
    symbol(optional) String

    Instrument symbol or currency to display in the order entry
    options(optional) Object

    Every attribute is optional. Price or Pips can only be specified for Limit and Stop orders.
    {
        type : "MARKET", "LIMIT" or "STOP",
        side : "BUY" or "SELL",
        price : Price Level,
        pips : Pip Distance
    }

    Opens the order entry light box in the center of the window - This method can be called when trading is not enabled too and an information message will be diplayed.

    tradableEmbed.promptOrderEntry = function(symbol, options) {
    	if(!$(".tr-embed-modal-overlay").length) {
			$("#trEmbedUIEnableTradingModal").append("<div class='tr-embed tr-embed-modal-overlay'></div>");
			$('.tr-embed-modal-overlay').transition({ opacity: .7 }, 100); // Using jQuery transit for CSS3 animations
    	}
    	
    	if(!tradableEmbed.tradingEnabled) {
    		orderEntry.adjustModalLocation();
    		splashModal.openEnableTradingModal();
    	} else {
    		orderEntry.openOrderEntryModal(symbol, options);
    	}
	};

///  Initialize UI  ///

function initializeUI() {
	console.log('Initializing tradableEmbedUI');

	notifications.initializeNotifications();
	
	splashModal.buildEnableTradingModal(responsiveMode);
	orderEntry.buildOrderEntryModal();
	tpSlPopup.buildTPSLPopup();
	tradeButtons.initializeTradeButtons();

	trackingHelper.trackPage();
	tradableEmbed.onEmbedReady(function () {
		console.log("Initializing Tradable Components... ");
		
		orderEntry.setCurrentModeOnPlaceOrderButton();

		if(tradableEmbed.tradingEnabled) {
			trackingHelper.initTracking();
			tradingEnabled();
		} else {
			scraper.clearLinks();
			tradingNotEnabled();
		}
		bottomBanner.buildBottomBanner(embedOptionsObj.showBottomBanner, responsiveMode);
		dashboard.buildDashboard();
		portfolioBuilder.buildPortfolioWidget();
		watchlist.buildWatchList();

		scraper.executePageScraper();
	});

	tradableEmbed.onError(function(err) {
		notifications.handleError(err.message);
	});
}

function tradingEnabled(){
	$(document).ready(function() {
		var trEnabled = localStorage.getItem("uiTradingEnabled");
		if(!trEnabled) {
			localStorage.setItem("uiTradingEnabled", "uiTradingEnabled");
		}
		trackingHelper.trackEvent('Trading Enabled');

		tradeButtons.initializeAuthTradeEnableButton();
		
		// Add widget update listener
		tradableEmbed.onAccountUpdated(function(account) {
			if(!!account) {
				if(dashboard.dashboardVertical !== undefined) {
					dashboard.setSnapshotValuesOnDashboard(account);
				}
				if(portfolioBuilder.portfolioVertical !== undefined) {
					portfolioBuilder.initializeOrUpdateRows(account);
				}
				if(watchlist.watchListVertical !== undefined) {
					watchlist.updatePricesOnWatchList(account.prices);
				}
				if(bottomBanner.bottomBannerShowing) {
					portfolioBuilder.initializeOrUpdateRows(tradableEmbed.lastSnapshot, true, "positions");
					portfolioBuilder.initializeOrUpdateRows(tradableEmbed.lastSnapshot, true, "orders");
				}
				bottomBanner.setSnapshotValuesOnBottomBanner(account);
			}
		});

		if(!responsiveMode) {
			// New user trade prompt
			notifications.scheduleNewUserTradePromptIfNeeded();
		}
	});
}

function tradingNotEnabled(){
        $(document).ready(function() {
        	tradeButtons.initializeUnauthTradeEnableButton();

		if($("#tr-embed-dashboard").length) {
			$("#tr-embed-dashboard").empty();
		}
		if($("#tr-embed-portfolio").length) {
			$("#tr-embed-portfolio").empty();
		}
		if($("#tr-embed-watch-list").length) {
			$("#tr-embed-watch-list").empty();
		}
        });
        orderEntry.stopQuotesTimer();
    }

/// jQuery extensions ///

$.fn.slideToggleWithCSS3 = function(callback) {
	if(!$(this).hasClass('tr-embed-transition')) {
		$(this).addClass('tr-embed-transition');
	}
	$(this).toggleClass('tr-embed-toggled');
	setTimeout(function() {
		if(!!callback && typeof callback == "function") {
			callback();
		}
	}, 100);
	return this;
};

$.fn.isToggledElementVisible = function() {
	return (!$(this).hasClass('tr-embed-toggled'));
}

$.fn.hideToggledElement = function() {
	if(!$(this).hasClass('tr-embed-toggled')) {
		$(this).addClass('tr-embed-toggled');
	}
}

    function injectCSS(id, path){
    	//"oldschool" way of doing it, which hopefully works accross all browsers. 
    	//In theory we could use setAttribute, unfortunately IE6 doesn't support it consistently.
	var cssId = id;
	if (!document.getElementById(cssId)){
		var head  = document.getElementsByTagName('head')[0];
		var link  = document.createElement('link');
		link.id   = cssId;
		link.rel  = 'stylesheet';
		link.type = 'text/css';
		link.href = path;
		link.media = 'all';
		head.appendChild(link);
	}
    }
